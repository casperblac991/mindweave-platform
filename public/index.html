async function generatePrompt() { const input = document.getElementById('labInput').value.trim(); if (!input) { alert('الرجاء كتابة وصف'); return; } const btn = document.getElementById('labGenerateBtn'); const output = document.getElementById('labOutput'); btn.disabled = true; btn.textContent = '⏳ جاري التوليد...'; output.textContent = ''; try { const response = await fetch('/api/generate', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ input: input }) }); const data = await response.json(); output.textContent = data.result || 'حدث خطأ'; document.getElementById('outputActions').style.display = 'block'; } catch (error) { output.textContent = '❌ خطأ: ' + error.message; } btn.disabled = false; btn.textContent = '⚡ توليد أمر احترافي'; }
